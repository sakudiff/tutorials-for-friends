% --- R Code Listings Configuration ---
% Supports: R, Python, Java, C++, MATLAB, SQL, and more
% Usage: \begin{lstlisting}[language=Python] ... \end{lstlisting}
\lstset{
  language=R,
  basicstyle=\small\ttfamily,
  breaklines=true,              % Auto-wrap long code lines
  breakatwhitespace=true,       % Break only at spaces (not mid-identifier)
  keepspaces=true,              % Preserve indentation on line breaks
  columns=flexible,             % Better spacing for monospace fonts
  showstringspaces=false,       % Do not mark spaces in strings
  keywordstyle=\color{blue},     % Keywords in blue
  commentstyle=\color{dlsugreen},% Comments in DLSU green
  stringstyle=\color{purple},    % Strings in purple
  backgroundcolor=\color{gray!5},% Light gray background
  frame=single,                  % Single-line border around code
  framerule=0pt,                 % No visible border (cleaner look)
  xleftmargin=10pt,              % Left padding inside code block
  xrightmargin=10pt,             % Right padding inside code block
  aboveskip=15pt,                % Space above code block
  belowskip=15pt,                % Space below code block
  breakindent=10pt,              % Indent amount for wrapped lines
  postbreak=\mbox{\textcolor{red}{$\hookrightarrow$}\space} % Arrow on broken lines
}

% --- APA 7th Edition: Headings & Indents ---
% Section headings are centered, bold, and large
% Subsection headings are left-aligned, bold, and medium
% Subsubsection headings are left-aligned, bold, and normal size
\titleformat{\section}{\normalfont\Large\bfseries\centering}{\thesection}{1em}{}
\titleformat{\subsection}{\normalfont\large\bfseries}{\thesubsection}{1em}{}
\titlespacing*{\section}{0pt}{12pt}{6pt}     % {left}{before}{after} spacing
\titlespacing*{\subsection}{0pt}{12pt}{6pt}
\titleformat{\subsubsection}{\normalfont\normalsize\bfseries}{\thesubsubsection}{1em}{}
\titlespacing*{\subsubsection}{0pt}{12pt}{6pt}
\setlength{\bibhang}{0.5in}  % Hanging indent for bibliography entries (APA style)

% --- Table Safety Configuration ---
% Prevents threeparttable footnotes from shrinking table column widths
% Tables are sized to content, not to the width of their footnotes
\renewcommand{\TPTnoteSettings}{%
  \parshape 1 0pt \linewidth
  \footnotesize\itshape
}

% --- Safe Table Environments ---
% \begin{safetable} ... \end{safetable}
%   Auto-scales any table that exceeds \textwidth to fit within margins.
%   Use this for all normal tables.
%
% \begin{largetable} ... \end{largetable}
%   Rotates the page to landscape and auto-scales the table.
%   Use this when a table has many columns and cannot fit portrait mode.
%
% Both environments center the table and work with threeparttable.
\newenvironment{safetable}[1][\textwidth]{%
  \begin{adjustbox}{max width=#1,center}%
}{%
  \end{adjustbox}%
}

\newenvironment{largetable}{%
  \begin{landscape}%
  \begin{adjustbox}{max width=\textwidth,center}%
}{%
  \end{adjustbox}%
  \end{landscape}%
}

% --- Float Placement Configuration ---
% Controls how LaTeX positions tables and figures on pages
% These values prioritize placing floats at the top of pages
\renewcommand{\topfraction}{0.9}       % Max fraction of page for top floats
\renewcommand{\bottomfraction}{0.9}    % Max fraction of page for bottom floats
\renewcommand{\textfraction}{0.1}      % Min fraction of page for text
\setcounter{topnumber}{3}              % Max floats at top of page
\setcounter{bottomnumber}{3}           % Max floats at bottom of page
\setcounter{totalnumber}{5}            % Max floats on a single page

% --- Orphan and Widow Control ---
% Prevents single lines stranded at page top (widow) or bottom (club)
% Setting penalties to 10000 makes LaTeX strongly avoid these situations
\clubpenalty=10000                     % Penalize single line at page bottom
\widowpenalty=10000                    % Penalize single line at page top
\displaywidowpenalty=10000             % Same for display math environments

% --- URL and Link Break Protection ---
% Allows long URLs and DOIs to break at numeric/letter boundaries
% Prevents these from overflowing into page margins
\urlstyle{same}                        % URLs use same font as body text
\setcounter{biburlnumpenalty}{100}     % Allow breaks after digits in URLs
\setcounter{biburlucpenalty}{100}      % Allow breaks after uppercase letters
\setcounter{biburllcpenalty}{100}      % Allow breaks after lowercase letters

% --- Math Equation Page Break Support ---
% Allows multi-line equation environments (align, gather) to span pages
% Without this, long equations cause "Underfull page" warnings
\allowdisplaybreaks

% --- Caption Formatting ---
% APA 7th edition requires single-spaced captions, smaller font
% Labels (Table 1, Figure 1) are bold
\captionsetup{font={small,singlespacing}, labelfont=bf}

% --- Figure Width Protection Command ---
% \safeincludegraphics{image.png} auto-scales images that exceed \textwidth
% Optional arguments work normally: \safeincludegraphics[width=3in]{image.png}
% Use this instead of \includegraphics to prevent image overflow
\newcommand{\safeincludegraphics}[2][]{%
  \adjustimage{max width=\textwidth, #1}{#2}%
}

% --- Bibliography Spacing Override ---
% APA 7th edition requires single-spaced references
% This overrides \doublespacing only within the bibliography
\AtBeginBibliography{\singlespacing}
